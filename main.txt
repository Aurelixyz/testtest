"""
main.py - Punkt wejścia aplikacji mSprzet (PyQt6)
"""
import sys
import ctypes
from PyQt6.QtWidgets import QApplication
from gui import Application

def force_light_titlebar(window):
    try:
        hwnd = int(window.winId())
        value = ctypes.c_int(0) # 0 = Jasny motyw, 1 = Ciemny motyw
        # Win 11 i nowsze kompilacje Win 10 (Build 19041+)
        ctypes.windll.dwmapi.DwmSetWindowAttribute(hwnd, 20, ctypes.byref(value), ctypes.sizeof(value))
        # Starsze kompilacje Win 10 (Build 17763 - 18363)
        ctypes.windll.dwmapi.DwmSetWindowAttribute(hwnd, 19, ctypes.byref(value), ctypes.sizeof(value))
    except Exception:
        pass

if __name__ == "__main__":
    app = QApplication(sys.argv)
    app.setStyle("Fusion") # Nowoczesny wygląd na każdym systemie
    window = Application()
    window.showMaximized()
    force_light_titlebar(window)
    sys.exit(app.exec())
