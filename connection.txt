"""
database/connection.py
Moduł zarządzający połączeniem z bazą danych SQL Server z automatycznym wyborem sterownika.
"""

import pyodbc
import re
from contextlib import contextmanager
from config import DB_SERVER, DB_DATABASE


def get_best_odbc_driver() -> str:
    """
    Wyszukuje i zwraca najnowszy zainstalowany sterownik ODBC dla SQL Server.
    """
    drivers = pyodbc.drivers()

    # 1. Szukaj sterowników z serii "ODBC Driver XX for SQL Server" (najbardziej optymalne)
    odbc_drivers = [d for d in drivers if re.match(r'^ODBC Driver \d+ for SQL Server$', d)]
    if odbc_drivers:
        # Sortuj malejąco po numerze wersji (np. 18, 17, 13)
        odbc_drivers.sort(key=lambda x: int(re.search(r'\d+', x).group()), reverse=True)
        return odbc_drivers[0]

    # 2. Fallback: Starszy "SQL Server Native Client"
    native_clients = [d for d in drivers if 'SQL Server Native Client' in d]
    if native_clients:
        native_clients.sort(reverse=True)
        return native_clients[0]

    # 3. Ostateczny fallback: Domyślny, najstarszy systemowy sterownik
    if 'SQL Server' in drivers:
        return 'SQL Server'

    raise pyodbc.Error("Błąd: Brak zainstalowanego jakiegokolwiek sterownika ODBC dla SQL Server w systemie!")


def create_connection() -> pyodbc.Connection:
    """
    Tworzy i zwraca połączenie z bazą danych używając najlepszego dostępnego sterownika.
    Rzuca wyjątek pyodbc.Error jeśli połączenie się nie uda.
    """
    best_driver = get_best_odbc_driver()

    conn_str = (
        f"Driver={{{best_driver}}};"
        f"Server={DB_SERVER};"
        f"Database={DB_DATABASE};"
        "Trusted_Connection=yes;"
        "TrustServerCertificate=yes;"
        "autocommit=True;"
    )
    return pyodbc.connect(conn_str)


@contextmanager
def get_connection():
    """
    Context manager do bezpiecznego zarządzania połączeniem.

    Użycie:
        with get_connection() as conn:
            cursor = conn.cursor()
            cursor.execute("SELECT ...")

    Automatycznie zamyka połączenie po wyjściu z bloku, nawet jeśli wystąpi wyjątek.
    """
    conn = None
    try:
        conn = create_connection()
        yield conn
    finally:
        if conn:
            try:
                conn.close()
            except Exception:
                pass
