from connection import create_connection

# Skrot dla czytelnosci - unikamy powtarzania pelnego schematu
S = DB_SCHEMA
U = f"{S}.[mSPRZET_KONTA_UZYTKOWNIKOW]"


def insert_record(table: str, data: dict):
    columns = ", ".join(f"[{key}]" for key in data.keys())
    placeholders = ", ".join("?" for _ in data)
    query = f"INSERT INTO {S}.[{table}] ({columns}) VALUES ({placeholders})"
    conn = create_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, tuple(data.values()))
        conn.commit()
    finally:
        conn.close()


def update_record(table: str, record_id: int, data: dict):
    # Wykrywamy, czy to tabela użytkowników, żeby użyć odpowiedniej nazwy kolumny ID
    id_col = "[ID_UZYTKOWNIKA]" if table == "mSPRZET_KONTA_UZYTKOWNIKOW" else "[ID]"

    set_clause = ", ".join(f"[{key}] = ?" for key in data.keys())
    query = f"UPDATE {S}.[{table}] SET {set_clause} WHERE {id_col} = ?"

    conn = create_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, tuple(data.values()) + (record_id,))
        conn.commit()
    finally:
        conn.close()


def delete_record_by_id(table: str, record_id: int):
    """
    Generyczna metoda do usuwania rekordu po ID.
    Wspiera kolumne ID dla wszystkich tabel.
    Dla tabeli uzytkownikow mozna uzyc tej samej logiki, jezeli pole nazywa sie ID (lub przekazac pole).
    W MSprzet_Konta_Uzytkownikow pole nazywa sie ID_UZYTKOWNIKA, wiec sprawdzamy to.
    """
    id_col = "[ID_UZYTKOWNIKA]" if table == "mSPRZET_KONTA_UZYTKOWNIKOW" else "[ID]"
    query = f"DELETE FROM {S}.[{table}] WHERE {id_col} = ?"
    conn = create_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, (record_id,))
        conn.commit()
    finally:
        conn.close()


def check_duplicate(table: str, conditions: list[tuple[str, str]]) -> bool:
    """
    Sprawdza, czy rekord o podanych parametrach juz istnieje.
    conditions to lista krotek (column_name, value). Zwraca True, jesli istnieje.
    Wymaga co najmniej jednego dopasowania (OR logic).
    """
    if not conditions:
        return False

    clauses = []
    values = []
    for col, val in conditions:
        if val:
            clauses.append(f"([{col}] = ? AND [{col}] IS NOT NULL AND [{col}] != '')")
            values.append(val)

    if not clauses:
        return False

    query = f"SELECT COUNT(*) FROM {S}.[{table}] WHERE " + " OR ".join(clauses)
    conn = create_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, tuple(values))
        return cursor.fetchone()[0] > 0
    finally:
        conn.close()

def search_users_advanced(search_values: dict) -> list:
    """
    Wyszukuje użytkowników na podstawie słownika kolumna -> wartosc.
    Ignoruje puste wartosci.
    """
    query_conditions = " AND ".join([f"[{key.strip('[]')}] LIKE ?" for key, val in search_values.items() if val])
    query_values = [f"%{val}%" for val in search_values.values() if val]

    query = f"""SELECT [NR KADROWY], IMIE, NAZWISKO, [RODZAJ ZATRUDNIENIA], LOKALIZACJA, ID_UZYTKOWNIKA 
               FROM {U}"""
    if query_conditions:
        query += " WHERE " + query_conditions

    conn = create_connection()
    try:
        cursor = conn.cursor()
        cursor.execute(query, query_values)
        return [list(row) for row in cursor.fetchall()]
    finally:
        conn.close()
