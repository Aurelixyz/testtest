class BackgroundWidget(QWidget):
    def __init__(self, parent_app=None):
        super().__init__()
        self.parent_app = parent_app

    def mousePressEvent(self, event):
        if self.parent_app and hasattr(self.parent_app, 'clear_all_selections'):
            self.parent_app.clear_all_selections()
        super().mousePressEvent(event)


class InteractiveTableWidget(QTableWidget):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        font = QFont("Arial", 10, QFont.Weight.Bold)
        self.horizontalHeader().setFont(font)
        self.horizontalHeader().setMinimumHeight(45)

    def mousePressEvent(self, event):
        item = self.itemAt(event.pos())
        if not item:
            self.clearSelection()
            self.clearFocus()
        super().mousePressEvent(event)

    def mouseMoveEvent(self, event):
        pass


class InteractiveTableView(QTableView):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        font = QFont("Arial", 10, QFont.Weight.Bold)
        self.horizontalHeader().setFont(font)
        self.horizontalHeader().setMinimumHeight(45)

    def mousePressEvent(self, event):
        item = self.indexAt(event.pos())
        if not item.isValid():
            self.clearSelection()
            self.clearFocus()
        super().mousePressEvent(event)

class OswiadczenieDateDialog(QDialog):
    """Specjalne okno dla Oświadczeń - Kalendarz + przyciski 1/15 (najbliższy dzień roboczy)."""

    def __init__(self, parent=None, title="Data złożenia oświadczenia"):

        layout.addWidget(self.calendar)

        # 2. Przyciski z inteligentnym wyliczaniem 1. i 15. dnia
        btn_layout = QHBoxLayout()

        btn_1.setStyleSheet("background-color: #87CEFA; font-weight: bold; padding: 6px;")
        btn_1.setCursor(Qt.CursorShape.PointingHandCursor)
        btn_1.clicked.connect(lambda: self.set_working_day_and_accept(1))

        btn_15 = QPushButton("Zatrudnienie 15")
        btn_15.setStyleSheet("background-color: #87CEFA; font-weight: bold; padding: 6px;")
        btn_15.setCursor(Qt.CursorShape.PointingHandCursor)
        btn_15.clicked.connect(lambda: self.set_working_day_and_accept(15))

        btn_layout.addWidget(btn_1)
        btn_layout.addWidget(btn_15)
        layout.addLayout(btn_layout)

        # 3. Zostawiamy tylko przycisk Anuluj
        btn_cancel = QPushButton("Anuluj")
        btn_cancel.setCursor(Qt.CursorShape.PointingHandCursor)
        btn_cancel.clicked.connect(self.reject)
        layout.addWidget(btn_cancel)

    def set_working_day_and_accept(self, target_day):
        import datetime
        from PyQt6.QtCore import QDate

        today = datetime.date.today()

        try:
            current_target = datetime.date(today.year, today.month, target_day)
        except ValueError:
            current_target = None

        if today.month == 12:
            next_target = datetime.date(today.year + 1, 1, target_day)
        else:
            next_target = datetime.date(today.year, today.month + 1, target_day)

        if current_target:
            best_target = current_target if abs((today - current_target).days) <= abs(
                (next_target - today).days) else next_target
        else:
            best_target = next_target

        for i in range(7):
            candidate = best_target + datetime.timedelta(days=i)
            if candidate.weekday() < 5:  # 0 to poniedziałek, 4 to piątek
                self.calendar.setSelectedDate(QDate(candidate.year, candidate.month, candidate.day))
                break

        self.accept()

    def get_date(self) -> str:
        return self.calendar.selectedDate().toString("yyyy-MM-dd")


class SearchUserDialog(QDialog):
    def __init__(self, parent):
        super().__init__(parent)
        self.setWindowTitle("Wyszukaj użytkownika")
        self.setFixedSize(900, 500)
        self.setModal(True)
        self.selected_user_data = None
        self.entries = {}
        self.filter_boxes = {}
        self.labels = ["NR KADROWY", "IMIE", "NAZWISKO", "RODZAJ ZATRUDNIENIA", "LOKALIZACJA", "ID_UZYTKOWNIKA"]
        self._build_ui()

    def _build_ui(self):
        layout = QVBoxLayout(self)

        self.filter_scroll = QScrollArea()
        self.filter_scroll.setWidgetResizable(True)
        self.filter_scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setVerticalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setFrameShape(QFrame.Shape.NoFrame)
        self.filter_scroll.setFixedHeight(60)

        self.filter_container = QWidget()
        self.filter_layout = QHBoxLayout(self.filter_container)
        self.filter_layout.setSpacing(0)
        self.filter_layout.setContentsMargins(0, 0, 0, 0)

        self.v_header_spacer = QWidget()
        self.filter_layout.addWidget(self.v_header_spacer)

        for col in self.labels:
            box = QWidget()
            box_layout = QVBoxLayout(box)
            box_layout.setContentsMargins(2, 0, 2, 0)
            box_layout.setSpacing(2)

            if col not in ("ID", "ID_UZYTKOWNIKA"):
                title_lbl = QLabel(DISPLAY_NAMES.get(col, col))
                title_lbl.setStyleSheet(
                    "font-size: 10px; color: gray; font-weight: bold; background: transparent; border: none;")
                box_layout.addWidget(title_lbl)

                entry = QLineEdit()
                entry.textChanged.connect(self._dynamic_search)
                self.entries[col] = entry
                box_layout.addWidget(entry)
            else:
                box_layout.addStretch()

            self.filter_layout.addWidget(box)
            self.filter_boxes[col] = box

        self.filter_layout.addStretch()
        self.filter_scroll.setWidget(self.filter_container)
        layout.addWidget(self.filter_scroll)

        self.table = InteractiveTableWidget()
        self.table.setColumnCount(len(self.labels))
        self.table.setHorizontalHeaderLabels([DISPLAY_NAMES.get(c, c) for c in self.labels])
        self.table.verticalHeader().setVisible(False)
        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        self.table.verticalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Fixed)

        for i, col_name in enumerate(self.labels):
            if col_name in ("ID", "ID_UZYTKOWNIKA"):
                self.table.horizontalHeader().setSectionResizeMode(i, QHeaderView.ResizeMode.ResizeToContents)

        self.table.setSelectionBehavior(QTableWidget.SelectionBehavior.SelectRows)
        self.table.setSelectionMode(QTableWidget.SelectionMode.SingleSelection)
        self.table.setEditTriggers(QTableWidget.EditTrigger.NoEditTriggers)
        self.table.itemDoubleClicked.connect(self._select_user)

        self.filter_layout.setContentsMargins(self.table.frameWidth(), 0, 0, 0)
        layout.addWidget(self.table)

        btn = QPushButton("Wybierz użytkownika")
        btn.setStyleSheet("background-color: lightgreen; font-weight: bold; padding: 10px;")
        btn.setCursor(Qt.CursorShape.PointingHandCursor)
        btn.clicked.connect(self._select_user)
        layout.addWidget(btn)

        self.table.horizontalHeader().sectionResized.connect(self._sync_filter_widths)
        self.table.horizontalHeader().geometriesChanged.connect(self._sync_filter_widths)
        self.table.horizontalScrollBar().valueChanged.connect(self.filter_scroll.horizontalScrollBar().setValue)

    def showEvent(self, event):
        super().showEvent(event)
        QTimer.singleShot(10, self._sync_filter_widths)

    def _sync_filter_widths(self, *args):
        if not hasattr(self, 'table'): return
        hw = self.table.verticalHeader().width() if self.table.verticalHeader().isVisible() else 0
        fw = self.table.frameWidth()
        self.v_header_spacer.setFixedWidth(hw + fw)

        for i, col in enumerate(self.labels):
            if col in self.filter_boxes:
                self.filter_boxes[col].setFixedWidth(self.table.columnWidth(i))

    def _dynamic_search(self):
        search_values = {lbl: entry.text() for lbl, entry in self.entries.items()}
        try:
            rows = db.search_users_advanced(search_values)
            self.table.setRowCount(len(rows))
            for r_idx, row in enumerate(rows):
                for c_idx, val in enumerate(row):
                    self.table.setItem(r_idx, c_idx, QTableWidgetItem(str(val) if val else ""))
        except Exception:
            pass

    def _select_user(self):
        selected = self.table.selectionModel().selectedRows()
        if not selected:
            QMessageBox.warning(self, "Błąd", "Nie wybrano użytkownika z listy.")
            return
        row = selected[0].row()
        self.selected_user_data = {self.labels[i]: self.table.item(row, i).text() for i in range(len(self.labels)) if
                                   self.table.item(row, i)}
        self.accept()

class DataTableModel(QAbstractTableModel):
    def __init__(self, data, headers):
        super().__init__()
        self._data = data
        self._headers = headers

    def data(self, index, role):
        if not index.isValid():
            return None
        if role == Qt.ItemDataRole.DisplayRole:
            val = self._data[index.row()][index.column()]
            return str(val) if val is not None else ""
        elif role == Qt.ItemDataRole.TextAlignmentRole:
            return Qt.AlignmentFlag.AlignLeft | Qt.AlignmentFlag.AlignVCenter
        return None

    def rowCount(self, index=QModelIndex()):
        return len(self._data)

    def columnCount(self, index=QModelIndex()):
        return len(self._headers)

    def headerData(self, section, orientation, role):
        if role == Qt.ItemDataRole.DisplayRole and orientation == Qt.Orientation.Horizontal:
            return self._headers[section]
        return None

    def update_data(self, new_data):
        self.beginResetModel()
        self._data = new_data
        self.endResetModel()

    def get_row_data(self, row_idx):
        return self._data[row_idx]


class MultiFilterProxyModel(QSortFilterProxyModel):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.filters = {}

    def set_filters(self, filters):
        self.filters = filters
        self.invalidateFilter()

    def filterAcceptsRow(self, source_row, source_parent):
        if not self.filters:
            return True
        for col_idx, f_val in self.filters.items():
            idx = self.sourceModel().index(source_row, col_idx, source_parent)
            data = self.sourceModel().data(idx, Qt.ItemDataRole.DisplayRole)
            if data and f_val not in str(data).lower():
                return False
        return True

                self.filter_entries[col] = entry
                box_layout.addWidget(entry)
            else:
                box_layout.addStretch()

            self.filter_layout.addWidget(box)
            self.filter_boxes[col] = box

        self.filter_layout.addStretch()
        self.filter_scroll.setWidget(self.filter_container)
        layout.addWidget(self.filter_scroll)

        # --- TABELA OPARTA NA MODELU I INTERAKTYWNYM SKALOWANIU CZCIONKI ---
        self.table = InteractiveTableView()
        self.table.setSelectionBehavior(QTableView.SelectionBehavior.SelectRows)
        self.table.setSelectionMode(QTableView.SelectionMode.SingleSelection)
        self.table.setEditTriggers(QTableView.EditTrigger.NoEditTriggers)
        self.table.setAlternatingRowColors(False)
        self.table.verticalHeader().setVisible(False)

        headers = [DISPLAY_NAMES.get(c, c) for c in self.columns]
        self.source_model = DataTableModel([], headers)
        self.proxy_model = MultiFilterProxyModel(self)
        self.proxy_model.setSourceModel(self.source_model)
        self.table.setModel(self.proxy_model)

        # --- CZYSTY STRETCH NAGŁÓWKÓW ---
        header = self.table.horizontalHeader()
        header.setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        for i, col_name in enumerate(self.columns):
            if col_name in ("ID", "ID_UZYTKOWNIKA"):
                header.setSectionResizeMode(i, QHeaderView.ResizeMode.ResizeToContents)

        self.table.selectionModel().selectionChanged.connect(
            lambda: self.parent_app.update_right_buttons(self.section_name))
        self.table.setContextMenuPolicy(Qt.ContextMenuPolicy.CustomContextMenu)
        self.table.customContextMenuRequested.connect(lambda pos: self.parent_app.show_context_menu(pos, self.table))

        self.filter_layout.setContentsMargins(self.table.frameWidth(), 0, 0, 0)
        layout.addWidget(self.table)

        self.lbl_count = QLabel("Licznik: 0")
        self.lbl_count.setStyleSheet(
            "font-weight: bold; font-size: 12px; padding: 2px; background: transparent; border: none;")
        layout.addWidget(self.lbl_count)

        self.table.horizontalHeader().sectionResized.connect(self._sync_filter_widths)
        self.table.horizontalHeader().geometriesChanged.connect(self._sync_filter_widths)
        self.table.horizontalScrollBar().valueChanged.connect(self.filter_scroll.horizontalScrollBar().setValue)

    def showEvent(self, event):
        super().showEvent(event)
        QTimer.singleShot(10, self._sync_filter_widths)

    def _sync_filter_widths(self, *args):
        if not hasattr(self, 'table'): return
        hw = self.table.verticalHeader().width() if self.table.verticalHeader().isVisible() else 0
        fw = self.table.frameWidth()
        self.v_header_spacer.setFixedWidth(hw + fw)

        for i, col in enumerate(self.columns):
            if col in self.filter_boxes:
                self.filter_boxes[col].setFixedWidth(self.table.columnWidth(i))

    def load_data(self):
        rows = self.fetch_method()
        self.source_model.update_data(rows)
        self.update_count()
        QTimer.singleShot(10, self._sync_filter_widths)

    def filter_data(self):
        active_filters = {}
        for col_name, entry in self.filter_entries.items():
            text = entry.text().strip().lower()
            if text:
                col_idx = self.columns.index(col_name)
                active_filters[col_idx] = text

        self.proxy_model.set_filters(active_filters)
        self.update_count()

    def update_count(self):
        visible = self.proxy_model.rowCount()
        self.lbl_count.setText(f"Licznik: {visible}")

    def get_selected_record_data(self):
        selected = self.table.selectionModel().selectedRows()
        if not selected: return None

        proxy_index = selected[0]
        source_index = self.proxy_model.mapToSource(proxy_index)

        row_data = self.source_model.get_row_data(source_index.row())
        return {self.columns[i]: str(row_data[i]) if row_data[i] is not None else "" for i in range(len(self.columns))}

    def handle_edit(self, section):
        if section == "UŻYTKOWNIK":
            self.edit_user_record_specific()
            return

        sec_widget = self.sections_widgets[section]
        data = sec_widget.get_selected_record_data()
        if not data: return

        id_key = "ID_UZYTKOWNIKA" if section == "KONTA UŻYTKOWNIKÓW" else "ID"
        record_id = int(data.get(id_key))
        self._open_record_dialog(section, record_id, data)

    def edit_user_record_specific(self):
        for sec_key, table in self.user_equipment_view.asset_tables.items():
            selected = table.selectionModel().selectedRows()
            if selected:
                row = selected[0].row()
                db_cols = table.db_columns
                data = {db_cols[i]: table.item(row, i).text() for i in range(table.columnCount()) if table.item(row, i)}
                record_id = int(data.get("ID"))

                cb = lambda: self.user_equipment_view.load_user_assets(self.user_values["ID_UZYTKOWNIKA"].text())
                self._open_record_dialog(sec_key, record_id, data, custom_callback=cb)
                return
        QMessageBox.warning(self, "Brak wyboru", "Zaznacz sprzęt do edycji w podglądzie.")

    def show_context_menu(self, position, table):
        menu = QMenu()
        copy_action = menu.addAction("Kopiuj Wiersz(e)")
        action = menu.exec(table.viewport().mapToGlobal(position))
        if action == copy_action:
            self.copy_selected_item(table)

    def copy_selected_item(self, table):
        model = table.model()
        selection = table.selectionModel().selectedRows()
        if not selection: return

        headers = []
        for i in range(model.columnCount()):
            h_data = model.headerData(i, Qt.Orientation.Horizontal, Qt.ItemDataRole.DisplayRole)
            # Usuwamy entery przy kopiowaniu, aby Excel łapał to jako jeden ciąg
            h_clean = str(h_data).replace('\n', ' ') if h_data else ""
            headers.append(h_clean)

        rows_text = ""
        for r_idx in selection:
            row_data = []
            for c in range(model.columnCount()):
                idx = model.index(r_idx.row(), c)
                d = model.data(idx, Qt.ItemDataRole.DisplayRole)
                row_data.append(str(d) if d else "")
            rows_text += "\t".join(row_data) + "\n"

        full_text = "\t".join(headers) + "\n" + rows_text.strip()
        QApplication.clipboard().setText(full_text)

    def generuj_oswiadczenie_laptop(self):
        user_id = self.user_values.get("ID_UZYTKOWNIKA").text()
        if user_id == "-":
            QMessageBox.warning(self, "Błąd", "Brak danych użytkownika.")
            return

        laptopy = db.get_laptopy_uzytkownika(int(user_id))
        if not laptopy:
            QMessageBox.warning(self, "Błąd", "Użytkownik nie ma przypisanego laptopa.")
            return

        # Zmiana: Szukamy, czy użytkownik zaznaczył konkretny laptop na liście
        selected_laptop = None
        if "LAPTOPY" in self.user_equipment_view.asset_tables:
            laptop_table = self.user_equipment_view.asset_tables["LAPTOPY"]
            selected_rows = laptop_table.selectionModel().selectedRows()
            if selected_rows:
                row = selected_rows[0].row()
                selected_laptop = {
                    "STATUS": laptop_table.item(row, 0).text() if laptop_table.item(row, 0) else "",
                    "NR_SRODKA_TRWALEGO": laptop_table.item(row, 1).text() if laptop_table.item(row, 1) else "",
                    "NAZWA_LAPTOPA": laptop_table.item(row, 2).text() if laptop_table.item(row, 2) else "",
                    "NR_SERYJNY": laptop_table.item(row, 3).text() if laptop_table.item(row, 3) else "",
                    "MODEL": laptop_table.item(row, 4).text() if laptop_table.item(row, 4) else "",
                    "NR_SDJ": laptop_table.item(row, 5).text() if laptop_table.item(row, 5) else ""
                }

        # Jeśli nie zaznaczył, a ma więcej niż 1 - wymuszamy wybór
        if not selected_laptop:
            if len(laptopy) > 1:
                QMessageBox.warning(self, "Wymagany wybór",
                                    "Użytkownik ma przypisany więcej niż jeden laptop. Zaznacz odpowiedni sprzęt na liście przed wygenerowaniem oświadczenia.")
                return
            else:
                # Jak ma tylko 1, bierzemy domyślnie ten pierwszy
                l = laptopy[0]
                selected_laptop = {"STATUS": l[0], "NR_SRODKA_TRWALEGO": l[1], "NAZWA_LAPTOPA": l[2],
                                   "NR_SERYJNY": l[3], "MODEL": l[4], "NR_SDJ": l[5]}

        dialog = OswiadczenieDateDialog(self, "Data złożenia oświadczenia")
        if dialog.exec() == QDialog.DialogCode.Accepted:
            try:
                _svc_oswiadczenie({k: v.text() for k, v in self.user_values.items()}, selected_laptop,
                                  dialog.get_date())
            except Exception as e:
                QMessageBox.critical(self, "Błąd", f"Nie udało się wygenerować dokumentu:\n{e}")

            unique_keys = ["NR SERYJNY"]
        elif section == "TELEFONY":
            unique_keys = ["IMEI", "NR TELEFONU"]
        elif section == "SŁUCHAWKI":

        file_path, _ = QFileDialog.getOpenFileName(self, "Wybierz plik", "", "Pliki Excel (*.xlsx *.xls)")
