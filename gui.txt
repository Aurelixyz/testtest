    def clear_all_selections(self):
        cleared = False
        if self.current_section in self.sections_widgets:
            t = self.sections_widgets[self.current_section].table
            if t.selectionModel().hasSelection():
                t.clearSelection()
                t.clearFocus()
                cleared = True

        if self.current_section == "UŻYTKOWNIK" and hasattr(self, 'user_equipment_view'):
            for t in self.user_equipment_view.asset_tables.values():
                if t.selectionModel().hasSelection():
                    t.clearSelection()
                    t.clearFocus()
                    cleared = True

        if cleared:
            self.update_right_buttons(self.current_section)

    def show_frame(self, section):
        for sec_widget in self.sections_widgets.values():
            sec_widget.table.clearSelection()
            sec_widget.table.clearFocus()
        for t in self.user_equipment_view.asset_tables.values():
            t.clearSelection()
            t.clearFocus()

        self.current_section = section

        for name, btn in self.tab_buttons.items():
            if name == section:
                btn.setObjectName("activeTab")
            else:
                btn.setObjectName("")
            btn.style().unpolish(btn)
            btn.style().polish(btn)

        if section == "UŻYTKOWNIK":
            self.btn_uzytkownik_tab.setObjectName("activeTab")
        else:
            self.btn_uzytkownik_tab.setObjectName("")
        self.btn_uzytkownik_tab.style().unpolish(self.btn_uzytkownik_tab)
        self.btn_uzytkownik_tab.style().polish(self.btn_uzytkownik_tab)

        if section == "UŻYTKOWNIK":
            self.user_info_frame.show()
            self.stacked_widget.setCurrentWidget(self.user_equipment_view)
        else:
            self.user_info_frame.hide()
            self.btn_uzytkownik_tab.hide()
            sec_widget = self.sections_widgets[section]
            self.stacked_widget.setCurrentWidget(sec_widget)
            sec_widget.load_data()

        self.update_right_buttons(section)

    def handle_show_history(self):
        section = self.current_section
        record_id = None
        table_name = None

        if section == "UŻYTKOWNIK":
            for sec_key, table in self.user_equipment_view.asset_tables.items():
                selected = table.selectionModel().selectedRows()
                if selected:
                    row = selected[0].row()
                    record_id = int(table.item(row, table.columnCount() - 1).text())
                    table_name = self.section_config[sec_key][0]
                    break
        elif section in self.sections_widgets:
            data = self.sections_widgets[section].get_selected_record_data()
            if data:
                id_key = "ID_UZYTKOWNIKA" if section == "KONTA UŻYTKOWNIKÓW" else "ID"
                record_id = int(data.get(id_key))
                table_name = self.section_config[section][0]

        if record_id and table_name:
            try:
                history_rows, db_cols = db.get_record_history(table_name, record_id)
                if not history_rows:
                    QMessageBox.information(self, "Brak danych",
                                            "Dla tego elementu nie zarejestrowano jeszcze żadnych zmian.")
                    return

                dlg = HistoryDialog(self, f"Rejestr zmian: {section} (ID: {record_id})", history_rows, db_cols)
                dlg.exec()
            except Exception as e:
                QMessageBox.critical(self, "Błąd", f"Nie udało się otworzyć historii:\n{e}")
        else:
            QMessageBox.warning(self, "Brak wyboru", "Zaznacz wiersz, aby wyświetlić historię zmian.")

    def ask_yes_no(self, title, message):
        msg_box = QMessageBox(self)
        msg_box.setWindowTitle(title)
        msg_box.setText(message)
        msg_box.setIcon(QMessageBox.Icon.NoIcon)

        btn_yes = msg_box.addButton("Tak", QMessageBox.ButtonRole.YesRole)
        btn_no = msg_box.addButton("Nie", QMessageBox.ButtonRole.NoRole)
        btn_yes.setCursor(Qt.CursorShape.PointingHandCursor)
        btn_no.setCursor(Qt.CursorShape.PointingHandCursor)
        msg_box.exec()
        return msg_box.clickedButton() == btn_yes

    def _get_actions_for_section(self, section):
        if section == "KONTA UŻYTKOWNIKÓW":
            return [
                ("POKAŻ SPRZĘT UŻYTKOWNIKA", self.otworz_szczegoly_uzytkownika),
                ("EDYTUJ KONTO", lambda: self.handle_edit(section)),
                ("USUŃ KONTO", lambda: self.handle_delete(section))
            ]
        elif section == "UŻYTKOWNIK":
            # --- ZMIANA: Rozbicie na dwa niezależne przyciski w menu bocznym ---
            return [
                ("EDYTUJ ZAZNACZONY SPRZĘT", lambda: self.handle_edit("UŻYTKOWNIK")),
                ("ODEPNIJ SPRZĘT", self.handle_unassign_from_user),
                ("USUŃ SPRZĘT", self.handle_delete_from_user),
                ("PRZYPISZ LAPTOP", lambda: self.handle_assign("LAPTOPY")),
                ("PRZYPISZ MONITOR", lambda: self.handle_assign("MONITORY")),
                ("PRZYPISZ TELEFON", lambda: self.handle_assign("TELEFONY")),
                ("PRZYPISZ SŁUCHAWKI", lambda: self.handle_assign("SŁUCHAWKI")),
                ("PRZYPISZ KARTĘ SIM", lambda: self.handle_assign("KARTY SIM")),
                ("PRZYPISZ ROUTER", lambda: self.handle_assign("ROUTER")),
                ("PRZYPISZ MYSZKĘ", lambda: self.handle_assign("MYSZKI")),
                ("PRZYPISZ KLAWIATURĘ", lambda: self.handle_assign("KLAWIATURY"))
            ]
        else:
            return [
                ("EDYTUJ", lambda: self.handle_edit(section)),
                ("USUŃ", lambda: self.handle_delete(section))
            ]

    def search_user_for_dialog(self):
        dialog = SearchUserDialog(self)
        if dialog.exec() == QDialog.DialogCode.Accepted and dialog.selected_user_data:
            return dialog.selected_user_data
        return None

    def otworz_szczegoly_uzytkownika(self):
        data = self.sections_widgets["KONTA UŻYTKOWNIKÓW"].get_selected_record_data()
        if not data:
            QMessageBox.information(self, "Brak wyboru", "Najpierw zaznacz użytkownika na liście.")
            return

        for k, v in self.user_values.items():
            v.setText(data.get(k, ""))
            v.setStyleSheet(
                "color: midnightblue; font-weight: bold; font-size: 11pt; background: transparent; border: none;")

        self.btn_uzytkownik_tab.show()
        self.show_frame("UŻYTKOWNIK")
        user_id = data.get("ID_UZYTKOWNIKA")
        self.user_equipment_view.load_user_assets(user_id)

    def handle_add(self, section):
        self._open_record_dialog(section, None, None)

    def handle_edit(self, section):
        if section == "UŻYTKOWNIK":
            self.edit_user_record_specific()
            return

        sec_widget = self.sections_widgets[section]
        data = sec_widget.get_selected_record_data()
        if not data: return

        id_key = "ID_UZYTKOWNIKA" if section == "KONTA UŻYTKOWNIKÓW" else "ID"
        record_id = int(data.get(id_key))
        self._open_record_dialog(section, record_id, data)

    def edit_user_record_specific(self):
        for sec_key, table in self.user_equipment_view.asset_tables.items():
            selected = table.selectionModel().selectedRows()
            if selected:
                row = selected[0].row()
                db_cols = table.db_columns
                data = {db_cols[i]: table.item(row, i).text() for i in range(table.columnCount()) if table.item(row, i)}
                record_id = int(data.get("ID"))

                cb = lambda: self.user_equipment_view.load_user_assets(self.user_values["ID_UZYTKOWNIKA"].text())
                self._open_record_dialog(sec_key, record_id, data, custom_callback=cb)
                return
        QMessageBox.warning(self, "Brak wyboru", "Zaznacz sprzęt do edycji w podglądzie.")

    def _open_record_dialog(self, section, record_id, initial_data, custom_callback=None):
        table_name = self.section_config[section][0]
        fields_config = self._get_fields_config(section)
        cb = custom_callback if custom_callback else lambda: self.sections_widgets[section].load_data()
        dlg = RecordDialog(self, f"{'Edytuj' if record_id else 'Dodaj'} - {section}", fields_config, table_name, [], cb,
                           record_id, initial_data)
        dlg.exec()

    def copy_selected_item(self, table):
        model = table.model()
        selection = table.selectionModel().selectedRows()
        if not selection: return

        headers = []
        for i in range(model.columnCount()):
            h_data = model.headerData(i, Qt.Orientation.Horizontal, Qt.ItemDataRole.DisplayRole)
            h_clean = str(h_data).replace('\n', ' ') if h_data else ""
            headers.append(h_clean)

        rows_text = ""
        for r_idx in selection:
            row_data = []
            for c in range(model.columnCount()):
                idx = model.index(r_idx.row(), c)
                d = model.data(idx, Qt.ItemDataRole.DisplayRole)
                row_data.append(str(d) if d else "")
            rows_text += "\t".join(row_data) + "\n"

        full_text = "\t".join(headers) + "\n" + rows_text.strip()
        QApplication.clipboard().setText(full_text)

    def generuj_oswiadczenie_laptop(self):
        user_id = self.user_values.get("ID_UZYTKOWNIKA").text()
        if user_id == "-":
            QMessageBox.warning(self, "Błąd", "Brak danych użytkownika.")
            return

        laptopy = db.get_laptopy_uzytkownika(int(user_id))
        if not laptopy:
            QMessageBox.warning(self, "Błąd", "Użytkownik nie ma przypisanego laptopa.")
            return

        selected_laptop = None
        if "LAPTOPY" in self.user_equipment_view.asset_tables:
            laptop_table = self.user_equipment_view.asset_tables["LAPTOPY"]
            selected_rows = laptop_table.selectionModel().selectedRows()
            if selected_rows:
                row = selected_rows[0].row()
                selected_laptop = {
                    "STATUS": laptop_table.item(row, 0).text() if laptop_table.item(row, 0) else "",
                    "NR_SRODKA_TRWALEGO": laptop_table.item(row, 1).text() if laptop_table.item(row, 1) else "",
                    "NAZWA_LAPTOPA": laptop_table.item(row, 2).text() if laptop_table.item(row, 2) else "",
                    "NR_SERYJNY": laptop_table.item(row, 3).text() if laptop_table.item(row, 3) else "",

                    "NR_SDJ": laptop_table.item(row, 5).text() if laptop_table.item(row, 5) else ""
                }

        if not selected_laptop:
            if len(laptopy) > 1:
                QMessageBox.warning(self, "Wymagany wybór",
                                    "Użytkownik ma przypisany więcej niż jeden laptop. Zaznacz odpowiedni sprzęt na liście przed wygenerowaniem oświadczenia.")
                return
            else:
                l = laptopy[0]
                selected_laptop = {"STATUS": l[0], "NR_SRODKA_TRWALEGO": l[1], "NAZWA_LAPTOPA": l[2],
                                   "NR_SERYJNY": l[3], "MODEL": l[4], "NR_SDJ": l[5]}

        if dialog.exec() == QDialog.DialogCode.Accepted:
            try:
                _svc_oswiadczenie({k: v.text() for k, v in self.user_values.items()}, selected_laptop,
                                  dialog.get_date())

    def import_from_excel(self, section):
        from excel_dialogs import ImportInfoDialog, GenericExcelImportPreviewDialog

        table_name, _, all_cols = self.section_config[section]
        db_cols = [c for c in all_cols if c not in ("ID", "ID_UZYTKOWNIKA", "NR KADROWY", "IMIE", "NAZWISKO")]

        unique_keys = []
        if section == "KONTA UŻYTKOWNIKÓW":
            db_cols = ["NR KADROWY", "IMIE", "NAZWISKO", "RODZAJ ZATRUDNIENIA", "LOKALIZACJA"]
            unique_keys = ["NR KADROWY"]
        elif section == "LAPTOPY":
            unique_keys = ["NR SRODKA TRWALEGO", "NR SERYJNY"]
        elif section == "MONITORY":
            unique_keys = ["NR SERYJNY"]
        elif section == "TELEFONY":
            unique_keys = ["IMEI", "NR TELEFONU"]
        elif section == "SŁUCHAWKI":
            unique_keys = ["NR SERYJNY"]
        elif section == "KARTY SIM":
            unique_keys = ["NR TELEFONU", "NR SIM"]
        elif section == "ROUTER":
            unique_keys = ["IMEI", "NR TELEFONU", "NR SIM"]
        elif section in ("MYSZKI", "KLAWIATURY"):
            unique_keys = []

        info_dlg = ImportInfoDialog(self, section, db_cols, unique_keys)
        if info_dlg.exec() != QDialog.DialogCode.Accepted:
            return

        file_path, _ = QFileDialog.getOpenFileName(self, "Wybierz plik", "", "Pliki Excel (*.xlsx *.xls)")
        if not file_path: return

        try:
            QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)
            existing_data = {}
            if section == "KONTA UŻYTKOWNIKÓW":
                existing_data = {"NR KADROWY": {str(r[0]).strip().upper() for r in db.get_all_konta() if r[0]}}
            elif section == "LAPTOPY":
                all_data = db.get_all_laptopy()
                existing_data = {
                    "NR SRODKA TRWALEGO": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR SERYJNY": {str(r[6]).strip().upper() for r in all_data if r[6]}
                }
            elif section == "MONITORY":
                all_data = db.get_all_monitory()
                existing_data = {"NR SERYJNY": {str(r[4]).strip().upper() for r in all_data if r[4]}}
            elif section == "TELEFONY":
                all_data = db.get_all_telefony()
                existing_data = {
                    "IMEI": {str(r[3]).strip().upper() for r in all_data if r[3]},
                    "NR TELEFONU": {str(r[9]).strip().upper() for r in all_data if r[9]}
                }
            elif section == "SŁUCHAWKI":
                all_data = db.get_all_sluchawki()
                existing_data = {"NR SERYJNY": {str(r[4]).strip().upper() for r in all_data if r[4]}}
            elif section == "KARTY SIM":
                all_data = db.get_all_karty_sim()
                existing_data = {
                    "NR TELEFONU": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR SIM": {str(r[5]).strip().upper() for r in all_data if r[5]}
                }
            elif section == "ROUTER":
                all_data = db.get_all_router()
                existing_data = {
                    "IMEI": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR TELEFONU": {str(r[5]).strip().upper() for r in all_data if r[5]},
                    "NR SIM": {str(r[6]).strip().upper() for r in all_data if r[6]}
                }

            df = pd.read_excel(file_path, dtype=str)
        except Exception as e:
            QApplication.restoreOverrideCursor()
            QMessageBox.critical(self, "Błąd pliku", str(e))
            return
        finally:
            QApplication.restoreOverrideCursor()

        expected_excel_cols = [c.lower().replace(" ", "_") for c in unique_keys]
        df_cols_lower = [str(c).lower().strip() for c in df.columns]
        missing_req = [req for req in expected_excel_cols if req not in df_cols_lower]

        if missing_req:
            QMessageBox.critical(self, "Błąd formatu",
                                 f"Brak kolumn w Excelu odpowiedzialnych za dane unikalne:\n{missing_req}")
            return

        dialog = GenericExcelImportPreviewDialog(self, df, section, db_cols, existing_data, unique_keys)
        if dialog.exec() == QDialog.DialogCode.Accepted:
            try:
                QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)
                for data in dialog.valid_data_to_import:
                    db.insert_record(table_name, data)
                self.sections_widgets[section].load_data()
            except Exception as e:
                QMessageBox.critical(self, "Błąd", str(e))
            finally:
                QApplication.restoreOverrideCursor()

    def closeEvent(self, event):
        for sec_widget in self.sections_widgets.values():
            if hasattr(sec_widget, 'worker') and sec_widget.worker and sec_widget.worker.isRunning():

                sec_widget.worker.wait(500)
