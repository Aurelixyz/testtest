class BackgroundWidget(QWidget):
    def __init__(self, parent_app=None):
        super().__init__()
        self.parent_app = parent_app

    def mousePressEvent(self, event):
        if self.parent_app and hasattr(self.parent_app, 'clear_all_selections'):
            self.parent_app.clear_all_selections()
        super().mousePressEvent(event)


class InteractiveTableWidget(QTableWidget):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.horizontalHeader().sectionResized.connect(self._adjust_header_font)

    def mousePressEvent(self, event):
        item = self.itemAt(event.pos())
        if not item:
            self.clearSelection()
            self.clearFocus()
        super().mousePressEvent(event)

    def mouseMoveEvent(self, event):
        pass  # Blokuje przeciąganie zaznaczenia po wierszach

    def resizeEvent(self, event):
        super().resizeEvent(event)
        self._adjust_header_font()

    def _adjust_header_font(self, *args):
        header = self.horizontalHeader()
        if not header or header.count() == 0 or header.sectionSize(0) == 0:
            return

        font = QFont("Arial", 11, QFont.Weight.Bold)
        best_size = 11
        for size in range(11, 6, -1):
            font.setPointSize(size)
            fm = QFontMetrics(font)
            fits = True
            for i in range(header.count()):
                text = self.model().headerData(i, Qt.Orientation.Horizontal)
                if text:
                    if fm.horizontalAdvance(str(text)) + 12 > header.sectionSize(i):
                        fits = False
                        break
            if fits:
                best_size = size
                break
        else:
            best_size = 7

        font.setPointSize(best_size)
        header.setFont(font)

class GenericExcelImportPreviewDialog(QDialog):
    """Zunifikowany podgląd importu. Sprawdza dane sprzętu lub kont."""

    def __init__(self, parent, dataframe, section, db_cols, existing_data, unique_keys):

        self._validate()

    def _build_ui(self):
        layout = QVBoxLayout(self)

        self.btn_cancel.clicked.connect(self.reject)

        self.btn_import.setStyleSheet("background-color: lightgreen; font-weight: bold;")

        self.btn_import.clicked.connect(self.accept)

        self.btn_layout.addWidget(self.btn_cancel)
        self.btn_layout.addWidget(self.btn_import)
        layout.addLayout(self.btn_layout)

    def _validate(self):
        df_col_map = {str(c).lower().strip(): str(c) for c in self.df.columns}
        expected_map = {c.lower().replace(" ", "_"): c for c in self.db_cols}

        seen_in_excel = {k: set() for k in self.unique_keys}
        green_bg, red_bg = QColor("#d4edda"), QColor("#f8d7da")

        self.table.setRowCount(len(self.df))
        for index, row in self.df.iterrows():
            row_data_dict = {}
            status, reason, bg_color = "OK", "", green_bg

            for excel_col_key, db_col in expected_map.items():
                val = ""
                if excel_col_key in df_col_map:
                    val = str(row.get(df_col_map[excel_col_key], "")).strip()
                    if val.lower() == "nan": val = ""
                row_data_dict[db_col] = val

            if "STATUS" in self.db_cols and not row_data_dict.get("STATUS"):
                row_data_dict["STATUS"] = "W magazynie"

            for u_key in self.unique_keys:
                val = row_data_dict.get(u_key, "")
                if not val:
                    continue
                val_upper = val.upper()
                if val_upper in self.existing_data.get(u_key, set()):
                    status, reason, bg_color, self.has_errors = "BŁĄD", f"{DISPLAY_NAMES.get(u_key, u_key)} istnieje w bazie", red_bg, True
                    break
                if val_upper in seen_in_excel[u_key]:
                    status, reason, bg_color, self.has_errors = "BŁĄD", f"Zduplikowany {DISPLAY_NAMES.get(u_key, u_key)} w pliku", red_bg, True
                    break
                seen_in_excel[u_key].add(val_upper)

            if self.section == "KONTA UŻYTKOWNIKÓW" and not row_data_dict.get("NR KADROWY"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak numeru kadrowego", red_bg, True
            elif self.section == "LAPTOPY" and not row_data_dict.get("NR SERYJNY") and not row_data_dict.get(
                    "NR SRODKA TRWALEGO"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak Numeru Seryjnego lub Środka Trw.", red_bg, True
            elif self.section in ("MONITORY", "SŁUCHAWKI") and not row_data_dict.get("NR SERYJNY"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak Numeru Seryjnego", red_bg, True
            elif self.section == "TELEFONY" and not row_data_dict.get("IMEI") and not row_data_dict.get("NR TELEFONU"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak IMEI lub Numeru Telefonu", red_bg, True
            elif self.section == "KARTY SIM" and not row_data_dict.get("NR TELEFONU") and not row_data_dict.get(
                    "NR SIM"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak Nr Telefonu lub Nr SIM", red_bg, True
            elif self.section == "ROUTER" and not row_data_dict.get("IMEI") and not row_data_dict.get("NR SIM"):
                status, reason, bg_color, self.has_errors = "BŁĄD", "Brak IMEI lub Nr SIM", red_bg, True

            if status == "OK":
                self.valid_data_to_import.append(row_data_dict)

            table_row = [status] + [row_data_dict[c] for c in self.db_cols] + [reason]
            for c_idx, v in enumerate(table_row):
                item = QTableWidgetItem(str(v))
                item.setBackground(bg_color)
                item.setForeground(QColor("black"))
                self.table.setItem(index, c_idx, item)

        if self.has_errors:
            self.info_lbl.setText("⚠ Wykryto błędy (wiersze na czerwono). Import zablokowany.")
            self.info_lbl.setStyleSheet("color: red; font-weight: bold; background: transparent;")
            self.btn_import.setEnabled(False)
            self.btn_import.setStyleSheet("background-color: lightgray; color: gray;")
        else:
            self.info_lbl.setText("✔ Wszystkie rekordy są poprawne.")
            self.info_lbl.setStyleSheet("color: green; font-weight: bold; background: transparent;")


class SearchUserDialog(QDialog):
    def __init__(self, parent):
        super().__init__(parent)
        self.setWindowTitle("Wyszukaj użytkownika")
        self.setFixedSize(900, 500)
        self.setModal(True)
        self.selected_user_data = None
        self.entries = {}
        self.filter_boxes = {}
        self.labels = ["NR KADROWY", "IMIE", "NAZWISKO", "RODZAJ ZATRUDNIENIA", "LOKALIZACJA", "ID_UZYTKOWNIKA"]
        self._build_ui()

    def _build_ui(self):
        layout = QVBoxLayout(self)

        self.filter_scroll = QScrollArea()
        self.filter_scroll.setWidgetResizable(True)
        self.filter_scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setVerticalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setFrameShape(QFrame.Shape.NoFrame)
        self.filter_scroll.setFixedHeight(50)

        self.filter_container = QWidget()
        self.filter_layout = QHBoxLayout(self.filter_container)
        self.filter_layout.setSpacing(0)
        self.filter_layout.setContentsMargins(0, 0, 0, 0)

        self.v_header_spacer = QWidget()
        self.filter_layout.addWidget(self.v_header_spacer)

        for col in self.labels:
            box = QWidget()
            box_layout = QVBoxLayout(box)
            box_layout.setContentsMargins(2, 0, 2, 0)
            box_layout.setSpacing(2)

            if col not in ("ID", "ID_UZYTKOWNIKA"):
                title_lbl = QLabel(DISPLAY_NAMES.get(col, col))
                title_lbl.setStyleSheet(
                    "font-size: 10px; color: gray; font-weight: bold; background: transparent; border: none;")
                box_layout.addWidget(title_lbl)

                entry = QLineEdit()
                entry.textChanged.connect(self._dynamic_search)
                self.entries[col] = entry
                box_layout.addWidget(entry)
            else:
                box_layout.addStretch()

            self.filter_layout.addWidget(box)
            self.filter_boxes[col] = box

        self.filter_layout.addStretch()
        self.filter_scroll.setWidget(self.filter_container)
        layout.addWidget(self.filter_scroll)

        self.table = InteractiveTableWidget()
        self.table.setColumnCount(len(self.labels))
        self.table.setHorizontalHeaderLabels([DISPLAY_NAMES.get(c, c) for c in self.labels])
        self.table.verticalHeader().setVisible(False)
        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        self.table.verticalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Fixed)

        for i, col_name in enumerate(self.labels):
            if col_name in ("ID", "ID_UZYTKOWNIKA"):
                self.table.horizontalHeader().setSectionResizeMode(i, QHeaderView.ResizeMode.ResizeToContents)

        self.table.setSelectionBehavior(QTableWidget.SelectionBehavior.SelectRows)
        self.table.setSelectionMode(QTableWidget.SelectionMode.SingleSelection)
        self.table.setEditTriggers(QTableWidget.EditTrigger.NoEditTriggers)
        self.table.itemDoubleClicked.connect(self._select_user)

        self.filter_layout.setContentsMargins(self.table.frameWidth(), 0, 0, 0)
        layout.addWidget(self.table)

        btn = QPushButton("Wybierz użytkownika")
        btn.setStyleSheet("background-color: lightgreen; font-weight: bold; padding: 10px;")
        btn.setCursor(Qt.CursorShape.PointingHandCursor)
        btn.clicked.connect(self._select_user)
        layout.addWidget(btn)

        self.table.horizontalHeader().sectionResized.connect(self._sync_filter_widths)
        self.table.horizontalHeader().geometriesChanged.connect(self._sync_filter_widths)
        self.table.horizontalScrollBar().valueChanged.connect(self.filter_scroll.horizontalScrollBar().setValue)

    def showEvent(self, event):
        super().showEvent(event)
        QTimer.singleShot(10, self._sync_filter_widths)

    def _sync_filter_widths(self, *args):
        if not hasattr(self, 'table'): return
        hw = self.table.verticalHeader().width() if self.table.verticalHeader().isVisible() else 0
        fw = self.table.frameWidth()
        self.v_header_spacer.setFixedWidth(hw + fw)

        for i, col in enumerate(self.labels):
            if col in self.filter_boxes:
                self.filter_boxes[col].setFixedWidth(self.table.columnWidth(i))

    def _dynamic_search(self):
        search_values = {lbl: entry.text() for lbl, entry in self.entries.items()}
        try:
            rows = db.search_users_advanced(search_values)
            self.table.setRowCount(len(rows))
            for r_idx, row in enumerate(rows):
                for c_idx, val in enumerate(row):
                    self.table.setItem(r_idx, c_idx, QTableWidgetItem(str(val) if val else ""))
        except Exception:
            pass

    def _select_user(self):
        selected = self.table.selectionModel().selectedRows()
        if not selected:
            QMessageBox.warning(self, "Błąd", "Nie wybrano użytkownika z listy.")
            return
        row = selected[0].row()
        self.selected_user_data = {self.labels[i]: self.table.item(row, i).text() for i in range(len(self.labels)) if
                                   self.table.item(row, i)}
        self.accept()

class UserEquipmentWidget(QScrollArea):
    def __init__(self, parent_app):
        super().__init__()
        self.parent_app = parent_app
        self.setWidgetResizable(True)
        self.setStyleSheet("QScrollArea { border: none; background-color: #f0f0f0; }")

        self.content_widget = BackgroundWidget(parent_app)
        self.layout = QVBoxLayout(self.content_widget)
        self.layout.setAlignment(Qt.AlignmentFlag.AlignTop)
        self.layout.setContentsMargins(10, 10, 10, 10)
        self.layout.setSpacing(10)
        self.setWidget(self.content_widget)

        self.asset_tables = {}

    def create_table_group(self, title, fetch_fn, user_id, cols, sec_key):
        group_box = QGroupBox(title)

        layout = QVBoxLayout(group_box)
        layout.setContentsMargins(8, 20, 8, 8)

        table = InteractiveTableWidget()
        table.db_columns = cols
        table.setColumnCount(len(cols))
        table.setHorizontalHeaderLabels([DISPLAY_NAMES.get(c, c) for c in cols])

        table.verticalHeader().setVisible(False)
        table.setAlternatingRowColors(True)

        table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        table.verticalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Fixed)

        for i, col_name in enumerate(cols):
            if col_name in ("ID", "ID_UZYTKOWNIKA"):
                table.horizontalHeader().setSectionResizeMode(i, QHeaderView.ResizeMode.ResizeToContents)

        table.setSelectionBehavior(QTableWidget.SelectionBehavior.SelectRows)
        table.setSelectionMode(QTableWidget.SelectionMode.SingleSelection)
        table.setEditTriggers(QTableWidget.EditTrigger.NoEditTriggers)

        try:
            rows = fetch_fn(int(user_id))
        except Exception:
            rows = []

        # ZAMRAŻAMY GUI I SYGNAŁY na czas wstawiania danych sprzętu użytkownika
        table.setUpdatesEnabled(False)
        table.blockSignals(True)
        table.setSortingEnabled(False)

        table.setRowCount(len(rows))
        for r_idx, row in enumerate(rows):
            for c_idx, val in enumerate(row):
                table.setItem(r_idx, c_idx, QTableWidgetItem(str(val) if val is not None else ""))

        # ODMRAŻAMY GUI
        table.blockSignals(False)
        table.setUpdatesEnabled(True)

        table.setContextMenuPolicy(Qt.ContextMenuPolicy.CustomContextMenu)
        table.customContextMenuRequested.connect(lambda pos, t=table: self.parent_app.show_context_menu(pos, t))
        table.itemSelectionChanged.connect(lambda s="UŻYTKOWNIK": self.parent_app.update_right_buttons(s))

        table.setMinimumHeight(max(70, 40 + (len(rows) * 30)))

        layout.addWidget(table)
        self.asset_tables[sec_key] = table
        return group_box

    def load_user_assets(self, user_id):
        for i in reversed(range(self.layout.count())):
            item = self.layout.itemAt(i)
            if item.widget():
                item.widget().deleteLater()
            elif item.layout():
                while item.layout().count():
                    child = item.layout().takeAt(0)
                    if child.widget():
                        child.widget().deleteLater()
                item.layout().deleteLater()

        self.asset_tables.clear()

class DataTableModel(QAbstractTableModel):
    def __init__(self, data, headers):
        super().__init__()
        self._data = data
        self._headers = headers

    def data(self, index, role):
        if not index.isValid():
            return None
        if role == Qt.ItemDataRole.DisplayRole:
            val = self._data[index.row()][index.column()]
            return str(val) if val is not None else ""
        elif role == Qt.ItemDataRole.TextAlignmentRole:
            return Qt.AlignmentFlag.AlignCenter
        return None

    def rowCount(self, index=QModelIndex()):
        return len(self._data)

    def columnCount(self, index=QModelIndex()):
        return len(self._headers)

    def headerData(self, section, orientation, role):
        if role == Qt.ItemDataRole.DisplayRole and orientation == Qt.Orientation.Horizontal:
            return self._headers[section]
        return None

    def update_data(self, new_data):
        self.beginResetModel()
        self._data = new_data
        self.endResetModel()

    def get_row_data(self, row_idx):
        return self._data[row_idx]

class MultiFilterProxyModel(QSortFilterProxyModel):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.filters = {}

    def set_filters(self, filters):
        self.filters = filters
        self.invalidateFilter()

    def filterAcceptsRow(self, source_row, source_parent):
        if not self.filters:
            return True
        for col_idx, f_val in self.filters.items():
            idx = self.sourceModel().index(source_row, col_idx, source_parent)
            data = self.sourceModel().data(idx, Qt.ItemDataRole.DisplayRole)
            if data and f_val not in str(data).lower():
                return False
        return True

class SectionWidget(BackgroundWidget):
    def __init__(self, section_name, fetch_method, columns, parent_app):
        super().__init__(parent_app)
        self.section_name = section_name
        self.fetch_method = fetch_method
        self.columns = columns
        self.filter_entries = {}
        self.filter_boxes = {}
        self._build_ui()

    def _build_ui(self):
        layout = QVBoxLayout(self)

        self.filter_scroll = QScrollArea()
        self.filter_scroll.setWidgetResizable(True)
        self.filter_scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setVerticalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.filter_scroll.setFrameShape(QFrame.Shape.NoFrame)
        self.filter_scroll.setFixedHeight(50)

        self.filter_container = QWidget()
        self.filter_layout = QHBoxLayout(self.filter_container)
        self.filter_layout.setSpacing(0)
        self.filter_layout.setContentsMargins(0, 0, 0, 0)

        self.v_header_spacer = QWidget()
        self.filter_layout.addWidget(self.v_header_spacer)

        for col in self.columns:
            box = QWidget()
            box_layout = QVBoxLayout(box)
            box_layout.setContentsMargins(2, 0, 2, 0)
            box_layout.setSpacing(2)

            if col not in ("ID", "ID_UZYTKOWNIKA"):
                lbl = QLabel(DISPLAY_NAMES.get(col, col))
                lbl.setStyleSheet(
                    "font-size: 10px; color: gray; font-weight: bold; background: transparent; border: none;")
                box_layout.addWidget(lbl)

                entry = QLineEdit()
                entry.textChanged.connect(self.filter_data)
                self.filter_entries[col] = entry
                box_layout.addWidget(entry)
            else:
                box_layout.addStretch()

            self.filter_layout.addWidget(box)
            self.filter_boxes[col] = box

        self.filter_layout.addStretch()
        self.filter_scroll.setWidget(self.filter_container)
        layout.addWidget(self.filter_scroll)

        # --- NOWA, LEKKA TABELA OPARTA NA MODELU ---
        self.table = QTableView()
        self.table.setSelectionBehavior(QTableView.SelectionBehavior.SelectRows)
        self.table.setSelectionMode(QTableView.SelectionMode.SingleSelection)
        self.table.setEditTriggers(QTableView.EditTrigger.NoEditTriggers)
        self.table.setAlternatingRowColors(False)
        self.table.verticalHeader().setVisible(False)

        # Inicjalizacja Modeli
        headers = [DISPLAY_NAMES.get(c, c) for c in self.columns]
        self.source_model = DataTableModel([], headers)
        self.proxy_model = MultiFilterProxyModel(self)
        self.proxy_model.setSourceModel(self.source_model)
        self.table.setModel(self.proxy_model)

        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)
        for i, col_name in enumerate(self.columns):
            if col_name in ("ID", "ID_UZYTKOWNIKA"):
                # USUNIĘTE MORDERDCZE DLA WYDAJNOŚCI ResizeToContents!
                self.table.horizontalHeader().setSectionResizeMode(i, QHeaderView.ResizeMode.Interactive)
                self.table.setColumnWidth(i, 50)

        self.table.selectionModel().selectionChanged.connect(
            lambda: self.parent_app.update_right_buttons(self.section_name))
        self.table.setContextMenuPolicy(Qt.ContextMenuPolicy.CustomContextMenu)
        self.table.customContextMenuRequested.connect(lambda pos: self.parent_app.show_context_menu(pos, self.table))

        self.filter_layout.setContentsMargins(self.table.frameWidth(), 0, 0, 0)
        layout.addWidget(self.table)

        self.lbl_count = QLabel("Licznik: 0")
        self.lbl_count.setStyleSheet(
            "font-weight: bold; font-size: 12px; padding: 2px; background: transparent; border: none;")
        layout.addWidget(self.lbl_count)

        self.table.horizontalHeader().sectionResized.connect(self._sync_filter_widths)
        self.table.horizontalHeader().geometriesChanged.connect(self._sync_filter_widths)
        self.table.horizontalScrollBar().valueChanged.connect(self.filter_scroll.horizontalScrollBar().setValue)

    def showEvent(self, event):
        super().showEvent(event)
        QTimer.singleShot(10, self._sync_filter_widths)

    def _sync_filter_widths(self, *args):
        if not hasattr(self, 'table'): return
        hw = self.table.verticalHeader().width() if self.table.verticalHeader().isVisible() else 0
        fw = self.table.frameWidth()
        self.v_header_spacer.setFixedWidth(hw + fw)

        for i, col in enumerate(self.columns):
            if col in self.filter_boxes:
                self.filter_boxes[col].setFixedWidth(self.table.columnWidth(i))

    def load_data(self):
        # Pobiera zawsze najnowsze dane i wrzuca do silnika, który natychmiast odświeża ekran!
        rows = self.fetch_method()
        self.source_model.update_data(rows)
        self.update_count()
        QTimer.singleShot(10, self._sync_filter_widths)

    def filter_data(self):
        active_filters = {}
        for col_name, entry in self.filter_entries.items():
            text = entry.text().strip().lower()
            if text:
                col_idx = self.columns.index(col_name)
                active_filters[col_idx] = text

        self.proxy_model.set_filters(active_filters)
        self.update_count()

    def update_count(self):
        visible = self.proxy_model.rowCount()
        self.lbl_count.setText(f"Licznik: {visible}")

    def get_selected_record_data(self):
        selected = self.table.selectionModel().selectedRows()
        if not selected: return None

        # Musimy zmapować indeks z proxy do bazowych danych
        proxy_index = selected[0]
        source_index = self.proxy_model.mapToSource(proxy_index)

        row_data = self.source_model.get_row_data(source_index.row())
        return {self.columns[i]: str(row_data[i]) if row_data[i] is not None else "" for i in range(len(self.columns))}

    def handle_edit(self, section):

            self.edit_user_record_specific()

    def edit_user_record_specific(self):
        for sec_key, table in self.user_equipment_view.asset_tables.items():
            selected = table.selectionModel().selectedRows()
            if selected:
                row = selected[0].row()
                db_cols = table.db_columns
                data = {db_cols[i]: table.item(row, i).text() for i in range(table.columnCount()) if table.item(row, i)}
                record_id = int(data.get("ID"))

                cb = lambda: self.user_equipment_view.load_user_assets(self.user_values["ID_UZYTKOWNIKA"].text())
                self._open_record_dialog(sec_key, record_id, data, custom_callback=cb)
                return
        QMessageBox.warning(self, "Brak wyboru", "Zaznacz sprzęt do edycji w podglądzie.")

    def handle_delete(self, section):
        sec_widget = self.sections_widgets[section]
        data = sec_widget.get_selected_record_data()
        if not data: return

        if self.ask_yes_no('Potwierdzenie', 'Czy na pewno chcesz bezpowrotnie usunąć ten rekord z bazy danych?'):

    def handle_delete_from_user(self):
        for sec_key, table in self.user_equipment_view.asset_tables.items():

            if selected:
                row = selected[0].row()
                record_id = int(table.item(row, table.columnCount() - 1).text())

                if self.ask_yes_no('Potwierdzenie usunięcia',
                                   f'Czy na pewno chcesz bezpowrotnie USUNĄĆ ten sprzęt z bazy danych?\n(Aby tylko odpiąć, użyj przycisku Edytuj)'):

                    self.user_equipment_view.load_user_assets(self.user_values["ID_UZYTKOWNIKA"].text())
                return
        QMessageBox.warning(self, "Brak wyboru", "Zaznacz sprzęt do usunięcia.")

    def show_context_menu(self, position, table):
        menu = QMenu()
        copy_action = menu.addAction("Kopiuj Wiersz(e)")
        action = menu.exec(table.viewport().mapToGlobal(position))
        if action == copy_action:
            self.copy_selected_item(table)

    def copy_selected_item(self, table):
        model = table.model()
        selection = table.selectionModel().selectedRows()
        if not selection: return

        headers = []
        for i in range(model.columnCount()):
            h_data = model.headerData(i, Qt.Orientation.Horizontal, Qt.ItemDataRole.DisplayRole)
            headers.append(str(h_data) if h_data else "")

        rows_text = ""
        for r_idx in selection:
            row_data = []
            for c in range(model.columnCount()):
                idx = model.index(r_idx.row(), c)
                d = model.data(idx, Qt.ItemDataRole.DisplayRole)
                row_data.append(str(d) if d else "")
            rows_text += "\t".join(row_data) + "\n"

        full_text = "\t".join(headers) + "\n" + rows_text.strip()
        QApplication.clipboard().setText(full_text)

    def generate_user_email(self):
        user_id = self.user_values.get("ID_UZYTKOWNIKA").text()
        user_data = {k: v.text() for k, v in self.user_values.items()}
        if user_id == "-": return

        try:
            equipment, col_desc, total, przelozony = _svc_get_equipment(user_id)
            if total == 0:
                QMessageBox.information(self, "Info", "Brak sprzętu u tego pracownika.")

            QMessageBox.critical(self, "Błąd", str(e))

    def import_from_excel(self, section):
        table_name, _, all_cols = self.section_config[section]
        db_cols = [c for c in all_cols if c not in ("ID", "ID_UZYTKOWNIKA", "NR KADROWY", "IMIE", "NAZWISKO")]

        unique_keys = []

        if section == "KONTA UŻYTKOWNIKÓW":
            db_cols = ["NR KADROWY", "IMIE", "NAZWISKO", "RODZAJ ZATRUDNIENIA", "LOKALIZACJA"]
            unique_keys = ["NR KADROWY"]
        elif section == "LAPTOPY":
            unique_keys = ["NR SRODKA TRWALEGO", "NR SERYJNY"]
        elif section == "MONITORY":
            unique_keys = ["NR SERYJNY"]
        elif section == "TELEFONY":
            unique_keys = ["IMEI", "NR TELEFONU"]
        elif section == "SŁUCHAWKI":
            unique_keys = ["NR SERYJNY"]
        elif section == "KARTY SIM":
            unique_keys = ["NR TELEFONU", "NR SIM"]
        elif section == "ROUTER":
            unique_keys = ["IMEI", "NR TELEFONU", "NR SIM"]
        elif section in ("MYSZKI", "KLAWIATURY"):
            unique_keys = []

        info_dlg = ImportInfoDialog(self, section, db_cols, unique_keys)
        if info_dlg.exec() != QDialog.DialogCode.Accepted:
            return

        file_path, _ = QFileDialog.getOpenFileName(self, "Wybierz plik", "", "Pliki Excel (*.xlsx *.xls)")
        if not file_path: return

        try:
            QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)

            existing_data = {}
            if section == "KONTA UŻYTKOWNIKÓW":
                existing_data = {"NR KADROWY": {str(r[0]).strip().upper() for r in db.get_all_konta() if r[0]}}
            elif section == "LAPTOPY":
                all_data = db.get_all_laptopy()
                existing_data = {
                    "NR SRODKA TRWALEGO": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR SERYJNY": {str(r[6]).strip().upper() for r in all_data if r[6]}
                }
            elif section == "MONITORY":
                all_data = db.get_all_monitory()
                existing_data = {"NR SERYJNY": {str(r[4]).strip().upper() for r in all_data if r[4]}}
            elif section == "TELEFONY":
                all_data = db.get_all_telefony()
                existing_data = {
                    "IMEI": {str(r[3]).strip().upper() for r in all_data if r[3]},
                    "NR TELEFONU": {str(r[9]).strip().upper() for r in all_data if r[9]}
                }
            elif section == "SŁUCHAWKI":
                all_data = db.get_all_sluchawki()
                existing_data = {"NR SERYJNY": {str(r[4]).strip().upper() for r in all_data if r[4]}}
            elif section == "KARTY SIM":
                all_data = db.get_all_karty_sim()
                existing_data = {
                    "NR TELEFONU": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR SIM": {str(r[5]).strip().upper() for r in all_data if r[5]}
                }
            elif section == "ROUTER":
                all_data = db.get_all_router()
                existing_data = {
                    "IMEI": {str(r[4]).strip().upper() for r in all_data if r[4]},
                    "NR TELEFONU": {str(r[5]).strip().upper() for r in all_data if r[5]},
                    "NR SIM": {str(r[6]).strip().upper() for r in all_data if r[6]}
                }

            df = pd.read_excel(file_path, dtype=str)
        except Exception as e:
            QApplication.restoreOverrideCursor()
            QMessageBox.critical(self, "Błąd pliku", str(e))
            return
        finally:
            QApplication.restoreOverrideCursor()

        expected_excel_cols = [c.lower().replace(" ", "_") for c in unique_keys]
        df_cols_lower = [str(c).lower().strip() for c in df.columns]
        missing_req = [req for req in expected_excel_cols if req not in df_cols_lower]

        if missing_req:
            QMessageBox.critical(self, "Błąd formatu",
                                 f"Brak kolumn w Excelu odpowiedzialnych za dane unikalne:\n{missing_req}")
            return

        dialog = GenericExcelImportPreviewDialog(self, df, section, db_cols, existing_data, unique_keys)
        if dialog.exec() == QDialog.DialogCode.Accepted:
            try:
                QApplication.setOverrideCursor(Qt.CursorShape.WaitCursor)
                for data in dialog.valid_data_to_import:
                    db.insert_record(table_name, data)
                QMessageBox.information(self, "Sukces", f"Zaimportowano {len(dialog.valid_data_to_import)} rekordów.")
                self.sections_widgets[section].load_data()
            except Exception as e:
                QMessageBox.critical(self, "Błąd", str(e))
            finally:
                QApplication.restoreOverrideCursor()
