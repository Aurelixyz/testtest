from connection import create_connection
# Dodajemy COMPANY_NAME do importu
from config import DB_SCHEMA, COMPANY_NAME

_S = DB_SCHEMA
_C = COMPANY_NAME # Skrót dla czytelności w tekście

# ─── Pomocnicze: Zarządzanie oknami Word ───────────────────────────────────

def _close_existing_msprzet_docs():
    if not platform.system().lower().startswith("win"): return
    try:
        word = win32com.client.GetActiveObject("Word.Application")
        targets = ("protokol odbioru", "oswiadczenie laptop")
        for i in range(word.Documents.Count, 0, -1):
            doc = word.Documents.Item(i)
            if any(doc.Name.lower().startswith(prefix) for prefix in targets):
                doc.Close(SaveChanges=False)
    except Exception: pass

def open_doc_on_app_screen(path: str):
    if not platform.system().lower().startswith("win"):
        if platform.system() == "Darwin": subprocess.Popen(["open", path])
        else: subprocess.Popen(["xdg-open", path])
        return
    try:
        app_windows = gw.getWindowsWithTitle("mSprzęt")
        target_x, target_y = 0, 0
        if app_windows:
            win = app_windows[0]
            target_x, target_y = win.left, win.top

        word = win32com.client.Dispatch("Word.Application")
        word.Visible = True
        doc = word.Documents.Open(path)
        word.ActiveWindow.WindowState = 0
        word.Left, word.Top = target_x + 100, target_y + 100
        word.ActiveWindow.WindowState = 1
        hwnd = word.ActiveWindow.Hwnd
        ctypes.windll.user32.SetForegroundWindow(hwnd)
        word.Activate()
    except Exception: os.startfile(path)

# ─── Pomocnicze: Dane ──────────────────────────────────────────────────────

def _fetch_active(user_id: str, table: str, cols_sql: str, cols_names: list) -> list:
    conn = create_connection()
    try:
        cur = conn.cursor()
        cur.execute(f"SELECT {cols_sql} FROM {_S}.[{table}] WHERE [ID_UZYTKOWNIKA] = ? AND ([STATUS] IS NULL OR [STATUS] NOT IN ('Zwrocony','Oddany')) ORDER BY [ID] ASC", (user_id,))
        return [{cols_names[i]: (row[i] if row[i] is not None else "") for i in range(len(cols_names))} for row in cur.fetchall()]
    finally: conn.close()

def _has_active(user_id: str, table: str) -> bool:
    conn = create_connection()
    try:
        cur = conn.cursor()
        cur.execute(f"SELECT TOP 1 1 FROM {_S}.[{table}] WHERE [ID_UZYTKOWNIKA] = ? AND ([STATUS] IS NULL OR [STATUS] NOT IN ('Zwrocony','Oddany'))", (user_id,))
        return cur.fetchone() is not None
    finally: conn.close()

# ─── Protokół odbioru ──────────────────────────────────────────────────────

def generuj_protokol_odbioru(user: dict, *args, **kwargs) -> str:
    imie, nazwisko, nr_kadrowy, user_id = user.get("IMIE", ""), user.get("NAZWISKO", ""), user.get("NR KADROWY", ""), user.get("ID_UZYTKOWNIKA", "")

    laptopy = _fetch_active(user_id, "mSPRZET_LAPTOPY", "[NR SRODKA TRWALEGO], [NAZWA LAPTOPA], [NR SERYJNY], [MODEL]", ["NR_ST", "NAZWA", "NR_SER", "MODEL"])
    telefony = _fetch_active(user_id, "mSPRZET_TELEFONY", "[MODEL], [IMEI], [NR TELEFONU]", ["MODEL", "IMEI", "NR_TEL"])
    monitory = _fetch_active(user_id, "mSPRZET_MONITORY", "[MODEL], [NR SERYJNY]", ["MODEL", "NR_SERYJNY"])
    sluchawki = _fetch_active(user_id, "mSPRZET_SLUCHAWKI", "[MODEL]", ["MODEL"])
    has_mysz, has_klaw = _has_active(user_id, "mSPRZET_MYSZKI"), _has_active(user_id, "mSPRZET_KLAWIATURY")
    has_przewody = len(monitory) > 0

    doc = Document()
    sec = doc.sections[0]
    sec.top_margin = sec.bottom_margin = sec.left_margin = sec.right_margin = Cm(2.5)
    doc.styles["Normal"].font.name, doc.styles["Normal"].font.size = "Calibri", Pt(11)

    # Nagłówek z dynamicznym _C
    hdr_txt = f"Informacje Służbowe podmiotu z Grupy {_C} - objęte ochroną {_C} Group's entity Business information - protected"
    doc.sections[0].header.paragraphs[0].add_run(hdr_txt).font.size = Pt(8)

    doc.add_paragraph("PROTOKÓŁ ODBIORU ŚRODKÓW TRWAŁYCH I AKCESORIÓW", style='Normal').alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.paragraphs[-1].runs[0].bold = True

    doc.add_paragraph(f"{imie} {nazwisko}, {nr_kadrowy}").alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph("(imię i nazwisko, numer kadrowy)").alignment = WD_ALIGN_PARAGRAPH.CENTER

    def _sec(t, lines):
        p = doc.add_paragraph(); p.add_run(t).bold = True
        if not lines: doc.add_paragraph("—")
        else: [doc.add_paragraph(l) for l in lines]

    _sec("1. LAPTOP:", [f"{lp['NR_ST']}\n{lp['NAZWA']}\n{lp['NR_SER']}\n{lp['MODEL']}" for lp in laptopy])
    _sec("2. TELEFON:", [f"{t['MODEL']}\nIMEI {t['IMEI']}\nnr telefonu {t['NR_TEL']}" for t in telefony])
    _sec("3. MONITOR:", [f"{m['MODEL']}\nService Tag\n{m['NR_SERYJNY']}" for m in monitory])
    _sec("4. SŁUCHAWKI:", [f"Model - {s['MODEL']}" for s in sluchawki])
    _sec("5. DODATKOWE AKCESORIA:", [f"Myszka - {'TAK' if has_mysz else 'NIE'}", f"Klawiatura - {'TAK' if has_klaw else 'NIE'}", f"Przewody - {'TAK' if has_przewody else 'NIE'}"])

    # Potwierdzenie z dynamicznym _C
    confirm = f"Potwierdzam przyjęcie powyższego sprzętu udostępnionego przez {_C} SA i zobowiązuję się dbać o jego należyty stan oraz zwrócić go na żądanie pracodawcy w terminie 7 dni do pracodawcy. Powyższy sprzęt jest własnością {_C} SA."
    doc.add_paragraph(confirm).alignment = WD_ALIGN_PARAGRAPH.JUSTIFY

    doc.add_paragraph("\nPrzyjmujący …………………………………………………………………………………………………")
    doc.add_paragraph(f"{imie} {nazwisko}, {nr_kadrowy} (DATA I MIEJSCOWOŚĆ)")

    out_path = os.path.join(tempfile.gettempdir(), f"Protokol odbioru - {nazwisko} {imie}.docx")
    _close_existing_msprzet_docs()
    if os.path.exists(out_path):
        try: os.remove(out_path)
        except Exception: pass
    doc.save(out_path)
    open_doc_on_app_screen(out_path)
    return out_path


# ─── Oświadczenie laptopa ──────────────────────────────────────────────────

def generuj_oswiadczenie_laptop(user: dict, laptop: dict, data_str: str, *args, **kwargs) -> str:
    """Generuje oświadczenie majątkowe zgodne z wzorem image_95bbb3.jpg."""
    imie, nazwisko, nr_kadrowy = user.get("IMIE", ""), user.get("NAZWISKO", ""), user.get("NR KADROWY", "")
    model, nr_st, nr_sdj = laptop.get("MODEL", ""), laptop.get("NR_SRODKA_TRWALEGO", "—"), laptop.get("NR_SDJ", "—")

    doc = Document()
    sec = doc.sections[0]
    sec.top_margin = sec.bottom_margin = sec.left_margin = sec.right_margin = Cm(2.5)
    doc.styles["Normal"].font.name, doc.styles["Normal"].font.size = "Calibri", Pt(11)

    # 1. Nagłówek (Header)
    hdr_txt = f"Informacje Służbowe podmiotu z Grupy {_C} - objęte ochroną | {_C} Group's entity Business Information - protected"
    doc.sections[0].header.paragraphs[0].add_run(hdr_txt).font.size = Pt(8)

    # Odstęp pod nagłówkiem
    doc.add_paragraph()

    # 2. Tytuł (Wyśrodkowany, pogrubiony)
    p_title = doc.add_paragraph()
    p_title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    r_title = p_title.add_run(
        f"Oświadczenie o odpowiedzialności majątkowej za powierzony do użytkowania służbowego sprzęt stanowiący własność {_C} S.A.")
    r_title.bold = True

    # Odstęp pod tytułem
    doc.add_paragraph()

    # 3. Akapit wstępny
    p_intro = doc.add_paragraph(
        f"Oświadczam, że przyjmuję do używania z obowiązkiem zwrotu lub wyliczenia się sprzęt będący własnością {_C} S.A. i powierzony mi przez {_C} S.A. zgodnie z art. 124 Kp, tj.:")
    p_intro.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY

    # Odstęp pod wstępem
    doc.add_paragraph()

    # 4. Dane sprzętu (Pogrubione etykiety)
    def _add_field(label, value):
        p = doc.add_paragraph()
        p.add_run(f"{label}: ").bold = True
        p.add_run(str(value) if value else "—")

    _add_field("Nazwa sprzętu", model)
    _add_field("Numer środka trwałego", nr_st)
    _add_field("Numer zgłoszenia", nr_sdj)

    # Odstęp pod danymi sprzętu
    doc.add_paragraph()

    # 5. Główna klauzula prawna
    p_main = doc.add_paragraph()
    p_main.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p_main.add_run(
        f"Oświadczam, że znany jest mi stan techniczny ww. sprzętu i w przypadku utraty sprzętu lub szkody w nim zobowiązuję się do pokrycia powstałych szkód w pełnej wysokości, w tym do pokrycia kosztów naprawy lub – gdy naprawa okaże się niemożliwa – do zwrotu jego wartości na wskazane konto, do maksymalnej kwoty 700 zł. W przypadku niedokonania naprawy lub wpłaty w terminie wskazanym przez {_C}, nie krótszym niż 7 dni od daty otrzymania wezwania do naprawienia szkody, upoważniam pracodawcę do potrącenia kwoty szkody z mojego wynagrodzenia za pracę, a także z innych należnych mi świadczeń pieniężnych. Wartość sprzętu wyceniona zostanie zgodnie z przepisami wewnętrznymi obowiązującymi w {_C}. Przepisy te dostępne są dla wszystkich pracowników {_C} w intranecie.")

    # Odstęp pod klauzulą
    doc.add_paragraph()

    # 6. Klauzula o zwrocie
    p_exit = doc.add_paragraph(
        f"Jednocześnie oświadczam, że z zastrzeżeniem zapisów powyższych jestem obowiązany do zwrotu na rzecz {_C} S.A. powierzonego sprzętu w terminie wskazanym przez {_C} S.A. w przypadku zakończenia współpracy z {_C} S.A.")
    p_exit.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY

    # Odstęp pod klauzulą końcową
    doc.add_paragraph()

    # 7. Dane użytkownika
    _add_field("Data złożenia oświadczenia", data_str)
    _add_field("Imię i nazwisko", f"{imie} {nazwisko}")
    _add_field("Nr kadrowy", nr_kadrowy)

    # Odstęp przed podpisem
    doc.add_paragraph()

    # 8. Podpis
    p_sig = doc.add_paragraph()
    p_sig.add_run("Podpis pracownika …………………………………………………………………………………").bold = True

    # Zapis i obsługa okien
    out_path = os.path.join(tempfile.gettempdir(), f"Oswiadczenie laptop - {nazwisko} {imie}.docx")
    _close_existing_msprzet_docs()
    if os.path.exists(out_path):
        try:
            os.remove(out_path)
        except Exception:
            pass
    doc.save(out_path)
    open_doc_on_app_screen(out_path)
    return out_path
