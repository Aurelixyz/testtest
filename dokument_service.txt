"""
services/dokument_service.py

Logika generowania dokumentow Word (DOCX):
  - Protokol odbioru srodkow trwalych
  - Oswiadczenie laptopa

Wydzielona z klasy Application.
Funkcje przyjmuja gotowe dane (dicts), nie odwoluja sie do widgetow Tkinter.
"""

import os
import platform
import subprocess
import tempfile

from docx import Document
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.shared import Cm, Pt

from connection import create_connection
from config import DB_SCHEMA


_S = DB_SCHEMA


# ─── Pomocnicze: pobieranie aktywnego sprzetu uzytkownika ──────────────────

def _fetch_active(user_id: str, table: str, cols_sql: str, cols_names: list) -> list:
    """Zwraca liste slownikow dla aktywnych rekordow w danej tabeli."""
    conn = create_connection()
    try:
        cur = conn.cursor()
        cur.execute(
            f"""
            SELECT {cols_sql}
            FROM {_S}.[{table}]
            WHERE [ID_UZYTKOWNIKA] = ?
              AND ([STATUS] IS NULL OR [STATUS] NOT IN ('Zwrocony','Oddany'))
            ORDER BY [ID] ASC
            """,
            (user_id,)
        )
        return [
            {cols_names[i]: (row[i] if row[i] is not None else "")
             for i in range(len(cols_names))}
            for row in cur.fetchall()
        ]
    finally:
        conn.close()


def _has_active(user_id: str, table: str) -> bool:
    """Sprawdza czy uzytkownik ma co najmniej 1 aktywny rekord w tabeli."""
    conn = create_connection()
    try:
        cur = conn.cursor()
        cur.execute(
            f"""
            SELECT TOP 1 1
            FROM {_S}.[{table}]
            WHERE [ID_UZYTKOWNIKA] = ?
              AND ([STATUS] IS NULL OR [STATUS] NOT IN ('Zwrocony','Oddany'))
            """,
            (user_id,)
        )
        return cur.fetchone() is not None
    finally:
        conn.close()


def get_active_laptop_for_user(user_id: str) -> dict | None:
    """
    Zwraca dane aktywnego laptopa uzytkownika lub None.
    Jesli uzytkownik ma wiecej niz 1 aktywny laptop, zwraca None
    (wymaga recznego wyboru w GUI).
    """
    conn = create_connection()
    try:
        cur = conn.cursor()
        cur.execute(
            f"""
            SELECT [MODEL], [NR SRODKA TRWALEGO], [NR SDJ], [NAZWA LAPTOPA], [NR SERYJNY]
            FROM {_S}.[mSPRZET_LAPTOPY]
            WHERE [ID_UZYTKOWNIKA] = ?
              AND ([STATUS] IS NULL OR [STATUS] NOT IN ('Zwrocony','Oddany'))
            ORDER BY [ID] DESC
            """,
            (user_id,)
        )
        rows = cur.fetchall()
    finally:
        conn.close()

    if len(rows) == 1:
        r = rows[0]
        return {
            "MODEL":              r[0] or "",
            "NR_SRODKA_TRWALEGO": r[1] or "",
            "NR_SDJ":             r[2] or "",
            "NAZWA_LAPTOPA":      r[3] or "",
            "NR_SERYJNY":         r[4] or "",
        }
    return None  # 0 lub >1 aktywnych


# ─── Otwieranie DOCX w Wordzie ─────────────────────────────────────────────

def open_doc(path: str, open_docx_foreground_fn=None) -> None:
    """
    Otwiera plik DOCX.
    Jesli dostarczona jest funkcja open_docx_foreground — uzywa jej (WinAPI foreground).
    W przeciwnym razie uzywa os.startfile (Windows) lub subprocess (Mac/Linux).
    """
    opened = False
    if open_docx_foreground_fn and platform.system().lower().startswith("win"):
        try:
            opened = open_docx_foreground_fn(path, maximize=True, mark_unsaved=True)
        except Exception:
            opened = False

    if not opened:
        if platform.system().lower().startswith("win"):
            os.startfile(path)
        elif platform.system() == "Darwin":
            subprocess.Popen(["open", path])
        else:
            subprocess.Popen(["xdg-open", path])


# ─── Protokol odbioru ──────────────────────────────────────────────────────

def generuj_protokol_odbioru(
    user: dict,
    open_docx_foreground_fn=None,
) -> str:
    """
    Generuje protokol odbioru srodkow trwalych dla uzytkownika.

    Parametry:
      user: slownik z kluczami NR_KADROWY, IMIE, NAZWISKO, ID_UZYTKOWNIKA
      open_docx_foreground_fn: opcjonalna funkcja do otwarcia Worda w tle

    Zwraca sciezke do wygenerowanego pliku DOCX.
    """
    imie       = user.get("IMIE", "")
    nazwisko   = user.get("NAZWISKO", "")
    nr_kadrowy = user.get("NR KADROWY", "")
    user_id    = user.get("ID_UZYTKOWNIKA", "")

    # Pobierz aktywny sprzet z DB
    laptopy   = _fetch_active(user_id, "mSPRZET_LAPTOPY",
                              "[NR SRODKA TRWALEGO], [NAZWA LAPTOPA], [NR SERYJNY], [MODEL]",
                              ["NR_ST", "NAZWA", "NR_SER", "MODEL"])
    telefony  = _fetch_active(user_id, "mSPRZET_TELEFONY",
                              "[MODEL], [IMEI], [NR TELEFONU]",
                              ["MODEL", "IMEI", "NR_TEL"])
    monitory  = _fetch_active(user_id, "mSPRZET_MONITORY",
                              "[MODEL], [NR SERYJNY]",
                              ["MODEL", "NR_SERYJNY"])
    sluchawki = _fetch_active(user_id, "mSPRZET_SLUCHAWKI",
                              "[MODEL]",
                              ["MODEL"])

    has_mysz    = _has_active(user_id, "mSPRZET_MYSZKI")
    has_klaw    = _has_active(user_id, "mSPRZET_KLAWIATURY")
    has_przewody = len(monitory) > 0

    # Budowanie dokumentu
    doc = Document()
    sec = doc.sections[0]
    sec.top_margin = sec.bottom_margin = sec.left_margin = sec.right_margin = Cm(2.5)
    base = doc.styles["Normal"]
    base.font.name = "Calibri"
    base.font.size = Pt(11)

    # Naglowek
    hdr_text = "x x"
    hdr = doc.sections[0].header
    p_hdr = hdr.paragraphs[0] if hdr.paragraphs else hdr.add_paragraph()
    p_hdr.alignment = WD_ALIGN_PARAGRAPH.LEFT
    r_hdr = p_hdr.add_run(hdr_text)
    r_hdr.font.name = "Calibri"
    r_hdr.font.size = Pt(8)

    # Tytul
    p_title = doc.add_paragraph()
    r = p_title.add_run("PROTOKOL ODBIORU SRODKOW TRWALYCH i AKCESORIOW")
    r.bold = True
    r.font.size = Pt(12)
    p_title.alignment = WD_ALIGN_PARAGRAPH.CENTER

    # Dane osoby
    doc.add_paragraph(f"{imie} {nazwisko}, {nr_kadrowy}").alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph("(imie i nazwisko, numer kadrowy)").alignment = WD_ALIGN_PARAGRAPH.CENTER

    def _section_title(title):
        p = doc.add_paragraph()
        p.add_run(title).bold = True

    def _add_lines(lines):
        if not lines:
            doc.add_paragraph("—")
        else:
            for line in lines:
                doc.add_paragraph(line)

    _section_title("1. LAPTOP:")
    _add_lines([
        f"{lp.get('NR_ST','—')}\n{lp.get('NAZWA','—')}\n{lp.get('NR_SER','—')}\n{lp.get('MODEL','—')}"
        for lp in laptopy
    ])

    _section_title("2. TELEFON:")
    _add_lines([
        f"{t.get('MODEL','—')}\nIMEI {t.get('IMEI','—')}\nnr telefonu {t.get('NR_TEL','—')}"
        for t in telefony
    ])

    _section_title("3. MONITOR:")
    _add_lines([
        f"{m.get('MODEL','—')}\nService Tag\n{m.get('NR_SERYJNY','—')}"
        for m in monitory
    ])

    _section_title("4. SLUCHAWKI:")
    _add_lines([f"Model - {s.get('MODEL','—')}" for s in sluchawki])

    _section_title("5. DODATKOWE AKCESORIA:")
    doc.add_paragraph(f"Myszka - {'TAK' if has_mysz else 'NIE'}")
    doc.add_paragraph(f"Klawiatura - {'TAK' if has_klaw else 'NIE'}")
    doc.add_paragraph(f"Przewody - {'TAK' if has_przewody else 'NIE'}")

    confirm = "x x"
    p_conf = doc.add_paragraph(confirm)
    p_conf.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY

    doc.add_paragraph()
    p_sig = doc.add_paragraph()
    p_sig.add_run("Przyjmujacy ").bold = True
    p_sig.add_run("…………………………………………………………………………………………………")
    doc.add_paragraph(f"{imie} {nazwisko}, {nr_kadrowy} (DATA i MIEJSCOWOSC)")

    out_path = os.path.join(tempfile.gettempdir(), f"Protokol odbioru - {nazwisko} {imie}.docx")
    doc.save(out_path)

    open_doc(out_path, open_docx_foreground_fn)
    return out_path


# ─── Oswiadczenie laptopa ──────────────────────────────────────────────────

def generuj_oswiadczenie_laptop(
    user: dict,
    laptop: dict,
    data_str: str,
    open_docx_foreground_fn=None,
) -> str:
    """
    Generuje oswiadczenie laptopa.

    Parametry:
      user:   slownik z kluczami IMIE, NAZWISKO, NR_KADROWY
      laptop: slownik z kluczami MODEL, NR_SRODKA_TRWALEGO, NR_SDJ, NAZWA_LAPTOPA, NR_SERYJNY
      data_str: data zlozona w formacie YYYY-MM-DD (string)
      open_docx_foreground_fn: opcjonalna funkcja do otwarcia Worda w tle

    Zwraca sciezke do wygenerowanego pliku DOCX.
    """
    imie        = user.get("IMIE", "")
    nazwisko    = user.get("NAZWISKO", "")
    nr_kadrowy  = user.get("NR KADROWY", "")

    model       = laptop.get("MODEL", "")
    nr_st       = laptop.get("NR_SRODKA_TRWALEGO", "—")
    nr_sdj      = laptop.get("NR_SDJ", "—")
    nazwa_sprzetu = model or ""

    doc = Document()
    sec = doc.sections[0]
    sec.top_margin = sec.bottom_margin = sec.left_margin = sec.right_margin = Cm(2.5)
    style = doc.styles["Normal"]
    style.font.name = "Calibri"
    style.font.size = Pt(11)

    # Naglowek
    header_text = "X | Y"
    hdr = doc.sections[0].header
    p_hdr = hdr.paragraphs[0] if hdr.paragraphs else hdr.add_paragraph()
    p_hdr.text = header_text
    for run in p_hdr.runs:
        run.font.name = "Calibri"
        run.font.size = Pt(8)

    # Tytul
    p_title = doc.add_paragraph()
    r = p_title.add_run("X X.")
    r.bold = True
    r.font.size = Pt(11)
    p_title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph()

    # Akapit 1
    p1 = doc.add_paragraph()
    p1.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p1.add_run("X X.:")
    doc.add_paragraph()

    def add_label_value(label: str, value: str):
        par = doc.add_paragraph()
        par.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
        par.add_run(f"{label}: ")
        par.add_run(value if value else "—")

    add_label_value("Nazwa sprzetu", nazwa_sprzetu)
    add_label_value("Numer srodka trwalego", nr_st or "—")
    add_label_value("Numer zgloszenia", nr_sdj or "—")

    doc.add_paragraph()
    p2 = doc.add_paragraph()
    p2.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p2.add_run("X ")

    doc.add_paragraph()
    p3 = doc.add_paragraph()
    p3.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p3.add_run("x ")

    doc.add_paragraph()
    add_label_value("Data zlozenia oswiadczenia", data_str)
    add_label_value("Imie i nazwisko", f"{imie} {nazwisko}")
    add_label_value("Nr kadrowy", nr_kadrowy)

    doc.add_paragraph()
    par_sig = doc.add_paragraph()
    par_sig.add_run("Podpis pracownika ").bold = True
    par_sig.add_run("……………………………………………………………")

    safe_name = f"Oswiadczenie x - {nazwisko} {imie}.docx"
    out_path = os.path.join(tempfile.gettempdir(), safe_name)
    if os.path.exists(out_path):
        try:
            os.remove(out_path)
        except Exception:
            pass
    doc.save(out_path)

    open_doc(out_path, open_docx_foreground_fn)
    return out_path
